\caption{RakennaOktettipuu}
\perusalgoritmi

\Input{Joukko pistepilviä $P$} %joissa jokaisen pisteen sijainti on ilmaistu $x$, $y$ ja $z$-koordinaateilla ja väri $rgb$-koodauksella}
\Output{Pisteet sisältävän oktettipuun juurisolmu $s$}

\kommentti{Selvitetään ensin pilvien rajauslaatikkojen unioni}
$L \gets$ tyhjä rajauslaatikko\\
\For{Pistepilvi $pc \in P$} {
    \For{Piste $p \in pc$} {
        \If{$p \cap L = \emptyset$}{
            $L \gets L \cup p$
        }
    }
}

\kommentti{Lisätään pisteet oktettipuuhun}
$s \gets$ juurisolmu, jonka rajauslaatikko on $L$\\
\For{Pistepilvi $pc \in P$} {
    Sekoita $pc$:n pisteet satunnaiseen järjestykseen\\
    \For{Piste $p \in pc$} {
        LisääSolmuun($s, p$)
    }
}
\Return{$s$}
