\perusalgoritmi

\Input{Sisäkkäispistepuu $P$}

$b_{yleiskuva} \gets$ pistepuskuri, jota pidetään näytönohjaimen muistissa

\uIf{Ensimmäinen visualisointikerta}{
    \While{$b_{yleiskuva}$ ei ole täynnä}{
        \kommentti{Käydään puuta läpi taso kerrallaan}
        \For{Solmu $s \in P$} {
            Lisää solmun $s$ pisteet puskuriin $b_{yleiskuva}$
        }
    }    
}

Lähetä $b_{yleiskuva}$ näytönohjaimelle piirrettäväksi.

\kommentti{Loput pisteet visualisoidaan puskurivirralla}
$S \gets$ näkyvissä olevat puun solmut, joita ei lisätty puskuriin $b_{yleiskuva}$\\
Järjestä S ruudulle projisoidun koon mukaan\\
PiirräPuskurivirta(S)
